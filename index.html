▶ Ver query BigQuery — QUERY_SELLERS (MLM) 
WITH carrier_per_seller AS (
  SELECT
    sla.SHP_SENDER_ID,
    t.CARRIER_NAME,
    COUNT(*) AS cnt,
    ROW_NUMBER() OVER (
      PARTITION BY sla.SHP_SENDER_ID
      ORDER BY COUNT(*) DESC
    ) AS rn
  FROM `meli-bi-data.WHOWNER.BT_SOE_SHIPMENT_SELLER_SLA` sla
  JOIN `meli-bi-data.WHOWNER.LK_SHP_FLEX_TRANSPORTATION` t
    ON sla.SHP_SHIPMENT_ID = t.SHP_SHIPMENT_ID
  WHERE sla.SITE_ID = 'MLM'                          -- ← MLM
    AND DATE(sla.REAL_DATE) >= DATE_SUB(CURRENT_DATE(), INTERVAL 30 DAY)
    AND t.SIT_SITE_ID = 'MLM'                        -- ← MLM
  GROUP BY 1, 2
)
SELECT
  p.SHP_SENDER_ID                              AS seller_id,
  p.SHP_EXPECTED_ORIGIN_NODE_ID                AS nodo,
  p.SHP_STATUS                                 AS status,
  SAFE_CAST(p.SHP_FIRST_EFFECTIVENESS AS FLOAT64) AS perf,
  c.CARRIER_NAME                               AS carrier
FROM `meli-bi-data.WHOWNER.BT_SOE_SELLER_PERFORMANCE_INCENTIVES` p
LEFT JOIN carrier_per_seller c
  ON p.SHP_SENDER_ID = c.SHP_SENDER_ID AND c.rn = 1
WHERE p.SHP_SITE          = 'MLM'                    -- ← MLM
  AND p.SHP_PICKING_TYPE_ID = 'self_service'
  AND SAFE_CAST(p.SHP_ANALYZED_SHIPMENTS AS INT64) > 0
  AND DATE(p.SHP_RUN_DATE) >= DATE_SUB(CURRENT_DATE(), INTERVAL 2 DAY)
ORDER BY p.SHP_SENDER_ID
LIMIT 5000

WITH raw_counts AS (
  SELECT
    sla.SHP_SENDER_ID,
    t.CARRIER_NAME,
    DATE_TRUNC(DATE(sla.REAL_DATE), WEEK(MONDAY)) AS week_start,
    COUNT(*) AS cnt
  FROM `meli-bi-data.WHOWNER.BT_SOE_SHIPMENT_SELLER_SLA` sla
  JOIN `meli-bi-data.WHOWNER.LK_SHP_FLEX_TRANSPORTATION` t
    ON sla.SHP_SHIPMENT_ID = t.SHP_SHIPMENT_ID
  WHERE sla.SITE_ID = 'MLM'                          -- ← MLM
    AND DATE(sla.REAL_DATE) >= DATE_SUB(CURRENT_DATE(), INTERVAL 28 DAY)
    AND t.SIT_SITE_ID = 'MLM'                        -- ← MLM
  GROUP BY 1, 2, 3
),
carrier_per_seller AS (
  SELECT SHP_SENDER_ID, CARRIER_NAME, week_start,
    ROW_NUMBER() OVER (PARTITION BY SHP_SENDER_ID, week_start ORDER BY cnt DESC) AS rn
  FROM raw_counts
),
base AS (
  SELECT
    p.SHP_SENDER_ID,
    SAFE_CAST(p.SHP_FIRST_EFFECTIVENESS AS FLOAT64) AS perf,
    c.CARRIER_NAME,
    c.week_start
  FROM `meli-bi-data.WHOWNER.BT_SOE_SELLER_PERFORMANCE_INCENTIVES` p
  JOIN carrier_per_seller c
    ON p.SHP_SENDER_ID = c.SHP_SENDER_ID AND c.rn = 1
  WHERE p.SHP_SITE            = 'MLM'                -- ← MLM
    AND p.SHP_PICKING_TYPE_ID = 'self_service'
    AND SAFE_CAST(p.SHP_ANALYZED_SHIPMENTS AS INT64) > 0
    AND DATE(p.SHP_RUN_DATE) >= DATE_SUB(CURRENT_DATE(), INTERVAL 28 DAY)
)
SELECT
  COALESCE(CARRIER_NAME, '(Sin carrier)')  AS carrier,
  CAST(week_start AS STRING)               AS week_start,
  COUNTIF(perf >= 97)                      AS muy_buenos,
  COUNTIF(perf >= 90 AND perf < 97)       AS buenos,
  COUNTIF(perf < 90)                       AS muy_malos,
  COUNT(*)                                 AS total
FROM base
GROUP BY 1, 2
ORDER BY 1, 2

▶ Ver query BigQuery — QUERY_PERFECT_DELIVERY (MLM) 
WITH visitas AS (
  SELECT
    s.SIT_SITE_ID                                    AS site,
    s.SHP_SHIPMENT_ID,
    s.SHP_LEAD_TIME_DEVIATION_TYPE,
    t.CARRIER_NAME,
    COUNT(DISTINCT CASE
      WHEN e.SHP_SUB_STATUS = 'receiver_absent'
      THEN e.SHP_SHIPMENT_ID END)                    AS visitas_fallidas,
    -- MLM: 21h hora local Mexico_City = EXTRACT(HOUR FROM DATETIME(ts, 'America/Mexico_City')) >= 21
    COUNT(DISTINCT CASE
      WHEN e.SHP_SUB_STATUS = 'receiver_absent'
        AND EXTRACT(HOUR FROM DATETIME(e.SHP_CHECKPOINT_DATETIME, 'America/Mexico_City')) >= 21
      THEN e.SHP_SHIPMENT_ID END)                    AS visitas_fallidas_pos21h,
    MAX(CASE WHEN COALESCE(s.REASON,'') = 'buyer_rescheduled' THEN 1 ELSE 0 END) AS is_repro_buyer
  FROM `meli-bi-data.WHOWNER.DM_SHP_FLEX_SUMMARY` s
  LEFT JOIN `meli-bi-data.WHOWNER.LK_SHP_FLEX_SHIPMENT_EVENTS` e
    ON s.SHP_SHIPMENT_ID = e.SHP_SHIPMENT_ID
  LEFT JOIN `meli-bi-data.WHOWNER.LK_SHP_FLEX_TRANSPORTATION` t
    ON s.SHP_SHIPMENT_ID = t.SHP_SHIPMENT_ID
  WHERE s.SHP_PICKING_TYPE_ID = 'self_service'
    AND s.SIT_SITE_ID         = 'MLM'               -- ← MLM
    AND DATE(s.SHP_DATETIME_HANDLING_ID)
          BETWEEN DATE_SUB(CURRENT_DATE(), INTERVAL 30 DAY) AND CURRENT_DATE()
  GROUP BY 1, 2, 3, 4
)
SELECT
  COALESCE(CARRIER_NAME, '(Sin carrier)')            AS carrier,
  COUNT(*)                                            AS total_envios,
  COUNTIF(visitas_fallidas = 0
    AND SHP_LEAD_TIME_DEVIATION_TYPE != 'DELAY'
    AND is_repro_buyer = 0)                           AS perfect_delivery,
  COUNTIF(SHP_LEAD_TIME_DEVIATION_TYPE = 'DELAY')    AS con_atraso,
  COUNTIF(visitas_fallidas > 0)                       AS ausencia_receptor,
  COUNTIF(visitas_fallidas_pos21h > 0)                AS ausencia_receptor_pos21h,
  COUNTIF(is_repro_buyer = 1)                         AS repro_buyer,
  ROUND(
    COUNTIF(visitas_fallidas = 0
      AND SHP_LEAD_TIME_DEVIATION_TYPE != 'DELAY')
    / COUNT(*) * 100, 1)                              AS pct_perfect_delivery
FROM visitas
GROUP BY 1
ORDER BY total_envios DESC
LIMIT 50
